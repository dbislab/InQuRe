SQL
SELECT full_actor_name
FROM actors
WHERE EXISTS (SELECT *
              FROM film f JOIN actors_films a ON f.ID = a.film_ID
              WHERE title LIKE '%study%' AND a.actorID = actors.ID);

SQL
SELECT a.author_id, a.author_name, COUNT(b.book_id) as total_sales
FROM authors a JOIN book_orders b ON a.author_id = b.book_author
WHERE a.author_name LIKE 'M%' AND b.purchase_date > '2020-01-01'
GROUP BY a.author_id, a.author_name
ORDER BY total_sales DESC;
